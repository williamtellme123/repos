-- ----------------------------------------------
-- NEW REPOS
-- ----------------------------------------------
    1. go to gitHub
    2. create new repo
    3. add new key if needed
    4. cd repos
    5. git clone git@github.com:williamtellme123/R.git
    6. Edit README.md
    7. git add README.md
    8. git status -s
    9. git commit -m 'Added TOC'
    10. git log
    11. git remote add origin  git@github.com:williamtellme123/R.git
    12. git push origin master
